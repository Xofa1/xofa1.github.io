Venture into the heart of a North American national park as a lone fire lookout. Witnessing a distant plume of smoke, your routine surveillance takes a chilling turn as you stumble upon an abandoned campsite, devoid of its occupants. Traces of struggle hint at a sinister presence lurking within the dense forest. Driven by curiosity and concern, you embark on a harrowing journey to unravel the mystery behind the vanished hikers. Following a trail stained with blood, you are led to the ominous entrance of an forgotten bunker hidden beneath the wilderness. Descend into the darkness and face the horrors lurking beneath the surface as you navigate the bunker's eerie depths. Uncover the grim fate of the missing hikers, only to come face-to-face with a horror beyond comprehension - a ravenous Wendigo hungry for your soul. Armed only with your wits and a fading sense of hope, you must race against time to escape the clutches of this malevolent entity. Can you survive the Wendigo's relentless pursuit and emerge from the depths or will you become just another victim of the darkness that haunts these lands?

An immersive game experience developed for the purpose of a sound study exploring the influence of sound on player perception and behavior in games, referenced in my bachelor thesis.

Own Experience

Game Development

Wendigo Mountain was developed over the course of about 3 months as part of my bachelor thesis. I decided to make a horror game, because of the clear results in research on the effects of sound in games. Playing a horror game without sound is much less frightening. I opted for a first person game because they are evidently more immersive compared to third person games. A full playthrough of the game should not take longer than about 20 minutes, as the study required the subjects to play the game twice in a row. To meet time constraints and since I was working alone, I decided on a dark forest setting, that inherently invokes fear and requires minimal assets. I chose the Unreal Engine to familiarize myself with it's features like the gameplay framework, terrain system and behavior trees. I programmed gameplay, created 3D models and textures, and designed and implemented sound for the game.

Sound Design

My concept for the sound design of the game focuses on a a mix of realistic sounds to represent game objects in the game world, creating a more immersive experience, and ambient noises, that are used to create a sense of unease that progressively intensifies as the player advances through the game. Although the player is never in any real danger until encountering the wendigo in the bunker, the distant animal noises and sounds of moving branches and grass imply that something is lurking in the dark. Furthermore sound is utilized to aid the player in navigating the forest section of the game. Spatial Sounds, such as the static noise of a dropped walkie talkie, draw the player towards the path leading to the next part of the level. The sounds were created and implemented into the game using Reaper and the Wwise Unreal Integration.

Personal Highlights

  • Tile Puzzle

    To progress the game, the player must solve a puzzle twice. The first time to fix the power supply to the elevator leading to the the second level of the game. The second time, the puzzle must be solved within a time limit in order to leave the bunker. To solve the puzzle, the player must rotate the tiles on the board to create a connection from the top left tile to the bottom right one.

APuzzle::APuzzle()
{
	PrimaryActorTick.bCanEverTick = true;

	SetRootComponent(Root);

	Board = CreateDefaultSubobject(TEXT("Board"));

	Camera->SetupAttachment(Board);

// Called when the game starts or when spawned
void APuzzle::BeginPlay()
{
	Super::BeginPlay();
	
	UWorld* World = GetWorld();
	TileActors.SetNumZeroed(TileData.Num());
	
	for(int i = 0; iSpawnActor(TileData[i].Tile,FVector::ZeroVector,FRotator::ZeroRotator);
		Actor->AttachToActor(this,FAttachmentTransformRules::SnapToTargetNotIncludingScale);
			
		FIntPoint TilePosition = FIntPoint(i % TileAmount.X,i / TileAmount.X);
		FVector Location = FVector(TilePosition.X * 20.0,0,TilePosition.Y * -20.0);
		FRotator Rotation = FRotator(0,0,90);
			
		Actor->SetActorRelativeLocation(Location);
		Actor->SetActorRelativeRotation(Rotation);
		Actor->SetTileRotation(static_cast(FMath::RandRange(0,3)));
		TileActors[i] = Actor;
	}

}

void APuzzle::Tick(float DeltaTime)
{
	Super::Tick(DeltaTime);
	for(APuzzleTile* Actor : TileActors)
	{
		if(Actor)
		{
			for(FPuzzleTileConnection& TileConnection : Actor->Connections)
			{
				TileConnection.bIsActive = false;
			}
		}
	}
	bPuzzleSolved = UpdatePath(StartTile,StartInput,TileAmount-1,EndOutput);
}

APuzzleTile* APuzzle::GetPuzzleTileAt(int32 x, int32 y)
{
	if((x<0||x>=TileAmount.X)||(y<0||y>=TileAmount.Y))
	{
		return nullptr;
	}
	int32 index = y * TileAmount.X + x;
	return TileActors[index];
}

bool APuzzle::UpdatePath(const FIntPoint& CurrentTilePosition, uint8 CurrentInput, const FIntPoint& TargetTilePosition, uint8 TargetOutput)
{	
	APuzzleTile* CurrentTile = GetPuzzleTileAt(CurrentTilePosition.X,CurrentTilePosition.Y);
	if (!CurrentTile)
	{
		return false;
	}
	
	for (uint8 ConnectionIndex = 0; ConnectionIndex < CurrentTile->Connections.Num(); ++ConnectionIndex)
	{	
		
		uint8 RotatedConnection = CurrentTile->GetRotatedConnection(ConnectionIndex);
		if ((RotatedConnection & CurrentInput)==0)
		{
			continue;
		}
		
		if (CurrentTile->Connections[ConnectionIndex].bIsActive)
		{
			continue;
		}
		CurrentTile->Connections[ConnectionIndex].bIsActive = true;
			
		//Check if Puzzle Solved
		if (CurrentTilePosition == TargetTilePosition && (TargetOutput & RotatedConnection))
		{
			return true;
		}

	
		if(RotatedConnection & 0b1000)
		{
			if (UpdatePath(CurrentTilePosition + FIntPoint(-1,0), 0b00000010, TargetTilePosition, TargetOutput))
			{
				return true;
			}
		}
		if(RotatedConnection & 0b0100)
		{
			if (UpdatePath(CurrentTilePosition + FIntPoint(0,1), 0b00000001, TargetTilePosition, TargetOutput))
			{
				return true;
			}
		}
		if(RotatedConnection & 0b0010)
		{
			if (UpdatePath(CurrentTilePosition + FIntPoint(1,0), 0b00001000, TargetTilePosition, TargetOutput))
			{
				return true;
			}
		}
		if(RotatedConnection & 0b0001)
		{
			if (UpdatePath(CurrentTilePosition + FIntPoint(0,-1), 0b00000100, TargetTilePosition, TargetOutput))
			{
				return true;
			}
		}		
	}
	return false;
}

bool APuzzle::GetIsPuzzleSolved()
{
	return bPuzzleSolved;
}

void APuzzleTile::RotateTile()
{
	PuzzleClickSFX();
	
	int32 RotationAsInt = (static_cast(Rotation) + 1) % static_cast(EPuzzleTileRotation::Count);
	EPuzzleTileRotation NewRotation = static_cast(RotationAsInt);
	SetTileRotation(NewRotation);
}


void APuzzleTile::SetTileRotation(EPuzzleTileRotation NewRotation)
{
	int32 DeltaRotation =  static_cast(NewRotation) - static_cast(Rotation);
	Rotation = NewRotation;

	AddActorLocalRotation(FRotator(0,90*DeltaRotation,0));
	
}

uint8 APuzzleTile::GetRotatedConnection(uint8 Index)
{
	int32 ConnectionFlags = Connections[Index].Flags;
	uint8 ConnectionShiftedByRotation = ConnectionFlags << static_cast(Rotation);
    
	uint8 First4BitsMasked = (ConnectionShiftedByRotation & 0b00001111);
	uint8 ExcessBitsShifted = (ConnectionShiftedByRotation >> 4);
    
	return First4BitsMasked | ExcessBitsShifted;    
}
